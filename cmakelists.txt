cmake_minimum_required(VERSION 3.10)
project(FMPartitioner)

set(CMAKE_CXX_STANDARD 17)
set(SOURCES main.cpp)

# optimization flags for release build
if(MSVC)
    # MSVC optimization flags
    set(CMAKE_CXX_FLAGS_RELEASE "/O2 /Ob2 /Oi /Ot /GL /arch:AVX2 /fp:fast /GS- /EHsc- /GR- /Gy /Gw /Zc:inline")
else()
    # GCC/Clang optimization flags
    set(CMAKE_CXX_FLAGS_RELEASE "-O3 -march=native -funroll-loops -fomit-frame-pointer -ffast-math -fno-strict-aliasing -fno-exceptions -fno-rtti -flto -ffunction-sections -fdata-sections -Wl,--gc-sections")
endif()
add_executable(fm_partition ${SOURCES})
